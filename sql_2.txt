SELECT B.CATEGORYNAME, SUM(A.QUANTITY)
FROM 
(SELECT PRODUCTID , ODD.QUANTITY FROM ORDERS OD LEFT JOIN ORDERDETAILS ODD ON OD.ORDERID = ODD.ORDERID WHERE SUBSTR(ORDERDATE,0,5)= '1996' ) A,
(SELECT CG.CATEGORYNAME, PD.PRODUCTID FROM CATEGORIES CG LEFT JOIN PRODUCTS PD ON CG.CATEGORYID = PD.CATEGORYID ) B
WHERE A.PRODUCTID = B.PRODUCTID
GROUP BY B.CATEGORYNAME
